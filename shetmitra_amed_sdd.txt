────────────────────────────────────────────────────────────────
3.1 NEW TABLE: amed_readings
────────────────────────────────────────────────────────────────

CREATE TABLE amed_readings (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  plot_id uuid REFERENCES farm_plots(id),
  fetch_date date NOT NULL,
  crop_type_detected text,
  crop_type_confidence numeric,
  field_size_acres_amed numeric,
  sowing_date date,
  harvest_date_predicted date,
  growth_stage text,
  growth_stage_confidence numeric,
  irrigation_detected boolean,
  last_event text,
  last_event_date date,
  data_refresh_date date,
  use_mock boolean DEFAULT true,
  raw_response jsonb,
  created_at timestamptz DEFAULT now()
);

CREATE INDEX idx_amed_plot ON amed_readings(plot_id);
CREATE INDEX idx_amed_date ON amed_readings(fetch_date);

────────────────────────────────────────────────────────────────
3.2 NEW TABLE: amed_belt_data
────────────────────────────────────────────────────────────────

CREATE TABLE amed_belt_data (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),

  harvest_week_start date,

  estimated_volume_mt numeric,
  health_pct_good numeric,
  health_pct_moderate numeric,
  health_pct_stressed numeric,
  health_pct_critical numeric,
  data_refresh_date date,
  raw_response jsonb,

CREATE INDEX idx_belt_region_date

FEATURE 1 — Weekly harvest volume forecast
  amed_belt_volume_mt: estimated MT arriving next 1-4 weeks
  This is a LEADING indicator — predicts arrivals
  before they show up in Agmarknet

  High AMED forecast volume → arrivals spike 1-2 weeks later
  → price pressure → prices drop
  Low AMED forecast volume → tight supply
  → price support → prices hold or rise

FEATURE 2 — Regional health distribution
  amed_health_pct_good: % of belt farms in good health
  High good % → high quality → premium prices
  Low good % → quality issues → discount prices

  Is this season early/late vs 3-year average?
  Early season → first-to-market premium
  Late season → rush to sell before quality drops

FEATURE 4 — Fields actively harvesting
  amed_fields_harvesting: count harvesting this week
  Proxy for actual market supply pressure

6.1 HARVEST WINDOW PREDICTION (improved)
────────────────────────────────────────────────────────────────

CURRENT:
  harvest_window = today + (NDVI declining trend estimate)
  Accuracy: ±5-7 days

NEW WITH AMED:
  Primary: harvest_date_predicted from AMED
  Refined by: NDVI trend from Sentinel-2
  
  Logic:
  amed_harvest = amed_reading.harvest_date_predicted
  ndvi_adjustment = calculate_ndvi_offset(
    current_ndvi, ndvi_30day_trend)
  
  If NDVI declining normally:
    harvest_window_start = amed_harvest - 3 days
    harvest_window_end = amed_harvest + 4 days
  If NDVI declining faster than expected:
    harvest_window_start = amed_harvest - 7 days
    harvest_window_end = amed_harvest - 1 day
  If NDVI not yet declining:
    harvest_window_start = amed_harvest
    harvest_window_end = amed_harvest + 10 days
  
  harvest_source = 'amed_confirmed'
  harvest_confidence = amed_confidence x ndvi_quality_factor
  
  Accuracy target: ±2-3 days (vs current ±5-7 days)

      Set amed_health_pct_good =

      Set amed_season_week =

      season_total_mt = amed_history.estimated_total_volume_mt
      harvest_weeks = weeks between harvest_start and harvest_end
      weekly_base = season_total_mt / harvest_weeks
      
      Apply bell curve distribution:
        Peak week = harvest_peak_date week
        Weeks before peak: ramp up
        Weeks after peak: ramp down
      
      This creates realistic arrival curve
      not flat weekly average

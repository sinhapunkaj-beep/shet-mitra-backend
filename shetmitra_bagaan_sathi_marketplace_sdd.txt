3.3 JARDALU GI VERIFICATION MODULE

File: geo/gi_verifier.py

class GIVerifier:
  def verify_jardalu(self, plot_id: str) -> dict:
    plot = get_plot(plot_id)
    
    # Check 1: Is farm within GI zone?
    gi_bbox = variety_config['Jardalu']['gi_zone_bbox']
    in_zone = (
      gi_bbox['south'] <= plot.centroid_lat <=
        gi_bbox['north'] and
      gi_bbox['west'] <= plot.centroid_lng <=
        gi_bbox['east']
    )
    
    # Check 2: AMED confirms Mango crop
    amed = get_latest_amed_reading(plot_id)
    is_mango = (
      amed.crop_type_detected == 'Mango' and
      amed.crop_type_confidence >= 0.80
    )
    
    # Check 3: Farmer registered as Jardalu
    farmer = get_farmer(plot.farmer_id)
    is_jardalu = (
      farmer.current_crop_variety == 'Jardalu'
    )
    
    gi_eligible = in_zone and is_mango and is_jardalu
    
    return {
      'gi_eligible': gi_eligible,
      'in_gi_zone': in_zone,
      'amed_confirmed': is_mango,
      'variety_registered': is_jardalu,
      'certificate_ref': f"SKI-GI-{plot_id[:8]}"
        if gi_eligible else None,
      'premium_multiplier': 1.60
        if gi_eligible else 1.0
    }

  def get_gi_certificate_text(self,
      farmer_id: str, plot_id: str) -> str:
    result = self.verify_jardalu(plot_id)
    farmer = get_farmer(farmer_id)
    if result['gi_eligible']:
      return f"""
      JARDALU GI VERIFICATION CERTIFICATE
      Bagaan Sathi | Sahyadri Krushi Intelligence
      
      Farmer: {farmer.farmer_full_name}
      Village: {farmer.village}, {farmer.district}
      Certificate: {result['certificate_ref']}
      
      ✅ GI Zone: Confirmed (Godda/Bhagalpur belt)
      ✅ Crop: AMED satellite verified Mango
      ✅ Variety: Jardalu (farmer registered)
      
      This lot is eligible for Jardalu GI
      premium pricing.
      """
    return None

def run_daily_matching():
  """Match active trader requirements
     with available farmer lots."""

  requirements = get_active_requirements()
  
  for req in requirements:
    # Find matching lots
    candidate_lots = get_lots_matching(
      commodity=req.commodity,
      variety=req.variety,
      grade=req.grade,
      harvest_from=req.collection_from,
      harvest_to=req.collection_to,
      region_code=req.region_code
    )
    
    for lot in candidate_lots:
      # Calculate match score
      score = calculate_match_score(lot, req)
      
      if score >= 0.70:
        # Create match record
        match = create_match(lot, req, score)
        
        # Notify farmer
        notify_farmer_of_offer(
          farmer_id=lot.farmer_id,
          lot=lot,
          requirement=req,
          match_id=match.id
        )
        
        # Log match
        log_match_created(match)

def calculate_match_score(lot, req) -> float:
  score = 0.0
  weights = {
    'commodity': 0.30,
    'variety': 0.20,
    'grade': 0.15,
    'quantity': 0.15,
    'timing': 0.10,
    'location': 0.10
  }
  
  # Commodity match (mandatory)
  if lot.commodity != req.commodity:
    return 0.0
  score += weights['commodity']
  
  # Variety match
  if req.variety is None or \
     lot.variety == req.variety:
    score += weights['variety']
  elif lot.variety and req.variety and \
       lot.variety.lower() in \
       req.variety.lower():
    score += weights['variety'] * 0.5
  
  # Grade match
  if lot.grade_predicted in req.grade:
    score += weights['grade']
  
  # Quantity overlap
  if lot.quantity_kg_estimated >= \
     req.quantity_kg_min:
    score += weights['quantity']
  elif lot.quantity_kg_estimated >= \
       req.quantity_kg_min * 0.7:
    score += weights['quantity'] * 0.6
  
  # Timing overlap
  overlap = date_range_overlap(
    lot.harvest_date_from,
    lot.harvest_date_to,
    req.collection_from,
    req.collection_to
  )
  score += weights['timing'] * overlap
  
  # Location proximity
  if lot.farm_district == req.location_district:
    score += weights['location']
  elif lot.farm_state == req.location_state:
    score += weights['location'] * 0.5
  
  return round(score, 3)

def run_weekly_aggregation():
  """Group small farmer lots into
     tradeable aggregated lots."""
  
  for region in get_active_regions():
    for commodity in region.primary_crops:
      # Get next 3 weeks of harvest windows
      for week_start in get_next_3_weeks():
        week_end = week_start + 7 days
        
        # Find all lots in this harvest window
        lots = get_lots_in_window(
          region_code=region.region_code,
          commodity=commodity,
          harvest_from=week_start,
          harvest_to=week_end,
          status='AVAILABLE'
        )
        
        # Group by grade
        for grade in ['A', 'B']:
          grade_lots = [l for l in lots
                        if l.grade_predicted == grade]
          
          total_kg = sum(l.quantity_kg_estimated
                         for l in grade_lots)
          
          # Only aggregate if > 2 farms
          # and > 1000 kg total
          if len(grade_lots) >= 2 and \
             total_kg >= 1000:
            create_aggregated_lot(
              region_code=region.region_code,
              commodity=commodity,
              grade=grade,
              harvest_week_start=week_start,
              lots=grade_lots,
              total_kg=total_kg
            )
            
            # Notify traders about
            # aggregated lot
            broadcast_aggregated_lot_to_traders(
              region_code=region.region_code,
              commodity=commodity,
              grade=grade,
              total_kg=total_kg,
              harvest_week=week_start
            )

def run_plant_priority_queue():
  """Fill Tasgaon plant with farm supply
     before external traders get access."""
  
  # Check plant capacity this week
  plant_capacity = get_plant_weekly_capacity()
  already_booked = get_plant_bookings_this_week()
  remaining_capacity_kg = (
    plant_capacity - already_booked
  )
  
  if remaining_capacity_kg < 500:
    return  # Plant full this week
  
  # Find Grade A Maharashtra farms
  # harvesting this week
  farms = get_available_farms(
    region_code='MH',
    commodity='Grapes',
    harvest_this_week=True,
    grade_predicted='A',
    not_yet_matched=True
  )
  
  for farm in farms:
    if remaining_capacity_kg <= 0:
      break
    
    # Calculate economics
    processing_fee = get_rate(
      'Washing, Sorting & Cleaning')
    # e.g. ₹3/kg processing fee
    
    mandi_price = get_current_mandi_price(
      'Dry Grapes', 'Tasgaon APMC')
    plant_price = mandi_price * 1.12
    # Plant pays 12% above mandi for Grade A
    
    net_to_farmer = plant_price - processing_fee
    premium_vs_mandi = net_to_farmer - mandi_price
    
    if premium_vs_mandi > 0:
      # Send priority offer to farmer
      send_plant_priority_offer(
        farmer_id=farm.farmer_id,
        estimated_kg=farm.quantity_kg_estimated,
        plant_price=plant_price,
        processing_fee=processing_fee,
        net_price=net_to_farmer,
        premium=premium_vs_mandi
      )
      
      remaining_capacity_kg -= \
        farm.quantity_kg_estimated
